## 功能

- 集成 Excalidraw 库，用于创建绘图和图表。

</PackageInfo>

## 安装

```bash
npm install @udecode/plate-excalidraw
```

## 使用

```tsx
import { ExcalidrawPlugin } from '@udecode/plate-excalidraw/react';
import { SelectOnBackspace } from '@udecode/plate-select';

const plugins = [
  // ...其他插件
  ExcalidrawPlugin,
  SelectOnBackspace.configure({
    options: { query: { allow: [ExcalidrawPlugin.key] } },
  }),
];
```

## 插件

### `ExcalidrawPlugin`

Excalidraw 空元素插件。

## API

### `insertExcalidraw`

将 Excalidraw 元素插入到编辑器中。

<API name="insertExcalidraw">
<APIParameters>
  <APIItem name="props" type="NodeProps<TExcalidrawElement>" optional>
    Excalidraw 元素的属性。
  </APIItem>
  <APIItem name="options" type="InsertNodesOptions" optional>
    插入 Excalidraw 元素的选项。
  </APIItem>
</APIParameters>
</API>

## 钩子

### `useExcalidrawElement`

Excalidraw 组件的行为钩子。

<API name="useExcalidrawElement">
<APIState>
  <APIItem name="element" type="TExcalidrawElement">
    Excalidraw 元素。
  </APIItem>
  <APIItem name="libraryItems" type="LibraryItems" optional>
    在 Excalidraw 组件中显示的库项目。
    - **默认值:** `[]`
  </APIItem>
  <APIItem name="scrollToContent" type="boolean" optional>
    是否滚动到 Excalidraw 组件内部的内容。
    - **默认值:** `true`
  </APIItem>
</APIState>

<APIReturns type="object">
  <APIItem name="Excalidraw" type="any">
    Excalidraw 组件。
  </APIItem>
  <APIItem name="excalidrawProps" type="ExcalidrawProps">
    传递给 Excalidraw 组件的属性。
  </APIItem>
</APIReturns>
</API>
